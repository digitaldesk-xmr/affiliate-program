import { supabase } from './supabase-client.js';
import cookie from 'cookie';

export default async function handler(req, res) {
    if (req.method !== 'POST') return res.status(405).json({ error: 'Method not allowed' });

    const { orderId, orderAmount, customerEmail, productId } = req.body;
    const cookies = cookie.parse(req.headers.cookie || '');
    const clickId = cookies.affiliate_click_id;

    if (!clickId) return res.json({ success: false, reason: 'no_affiliate_cookie' });

    // Recupera click e commissione dell'affiliato
    let finalProductId = productId;
    let commissionRate = null;

    const { data: click, error: clickError } = await supabase
        .from('clicks')
        .select('affiliate_id, affiliate:affiliates(commission_rate), link:affiliate_links(product_id)')
        .eq('id', clickId)
        .single();

    if (clickError || !click) return res.json({ success: false, reason: 'invalid_click' });

    commissionRate = click.affiliate.commission_rate;
    if (!finalProductId && click.link?.product_id) finalProductId = click.link.product_id;

    // Se il prodotto esiste e ha una commissione specifica, usala
    if (finalProductId) {
        const { data: product } = await supabase
            .from('products')
            .select('default_commission_rate')
            .eq('id', finalProductId)
            .single();
        if (product) commissionRate = product.default_commission_rate;
    }

    const commission = orderAmount * (commissionRate / 100);

    const { error: convError } = await supabase
        .from('conversions')
        .insert([{
            affiliate_id: click.affiliate_id,
            click_id: clickId,
            product_id: finalProductId,
            order_id: orderId,
            amount: orderAmount,
            commission: commission,
            status: 'pending'
        }]);

    if (convError) return res.status(500).json({ error: convError.message });

    res.json({ success: true, commission, status: 'pending_approval' });
}
